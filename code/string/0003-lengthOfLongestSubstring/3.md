# 無重複字符的最長子串

## Version 1
雙指針，思路還是清晰的
```go
func lengthOfLongestSubstring(s string) int {
	result := [][]byte{}
	if len(s) <= 1 {
		return len(s)
	}

	flag := 0
	for i := 1; i < len(s); i++ {
		tmp := []byte{s[0]}
		for j := flag; j < i; j++ {
			if s[i] != s[j] {
				tmp = append(tmp, s[i])
			} else {
				flag = j + 1
				j = i
			}
			result = append(result, tmp)
		}
	}

	var max = 0
	for i := range result {
		if len(result[i]) > max {
			max = len(result[i])
		}
	}

	return max
}
```

## Version 2
優化了一下，實現 雙100% :drooling_face:
```go
func lengthOfLongestSubstring(s string) int {
	// 判定最基礎的情況
	if len(s) <= 1 {
		return len(s)
	}

	var result int
	// flag 用來調整 j，區間 [j：i] 是滑動窗口
	flag := 0
	tmp := make([]byte, 1)
	for i := 1; i < len(s); i++ {
        tmp[0] = s[0]
		for j := flag; j < i; j++ {
			if s[i] != s[j] {
				tmp = append(tmp, s[i])
			} else {
				flag = j + 1
				j = i
			}

            // 判斷 result 是否是 max
			if result < len(tmp) {
				result = len(tmp)
			}
		}
        // 類似初始化操作，Cap信息仍保存
		tmp = tmp[0:1]
	}

	return result
}
```
