# 詞典中最長的單詞

## Version 1

蛤蛤, 又是暴力

```go
func longestWord(words []string) (result string) {
	m := make(map[string]int)
	for i := range words {
		if _, ok := m[words[i]]; !ok {
			m[words[i]] = 1
		}
	}

	result = ""
	for _, w := range words {
		isVaild := true

		for i := range w {
			s := string(w[:i+1])
			if _, ok := m[s]; !ok {
				isVaild = false
				break
			}
		}

		if isVaild {
			if len(w) == len(result) {
				for i := range w {
					if w[i] < result[i] {
						result = w
						break
					} else if w[i] > result[i] {
						break
					}
				}
			}
			if len(w) > len(result) {
				result = w
			}
		}
	}

	return
}
```
